BigQuery Operators

BigQueryInsertJobOperator Operator

Use case: Run any BQ job (query, load, extract, copy). Most flexible BQ operator.

from airflow.providers.google.cloud.operators.bigquery import BigQueryInsertJobOperator

BigQueryInsertJobOperator(
    task_id="run_query",
    configuration={
        "query": {
            "query": "SELECT * FROM `project.dataset.table` WHERE date = '{{ ds }}'",
            "useLegacySql": False,
            "destinationTable": {
                "projectId": "my-project",
                "datasetId": "my_dataset",
                "tableId": "target_table"
            },
            "writeDisposition": "WRITE_TRUNCATE"
        }
    }
)

BigQueryGetDataOperator Operator

Use case: Fetch rows from a BQ table and push to XCom for downstream tasks.

from airflow.providers.google.cloud.operators.bigquery import BigQueryGetDataOperator

BigQueryGetDataOperator(
    task_id="get_data",
    dataset_id="my_dataset",
    table_id="config_table",
    max_results=100,
    selected_fields="col1,col2"
)
# Access in downstream: ti.xcom_pull(task_ids='get_data')

BigQueryCheckOperator Operator

Use case: Data quality check – fails if query returns no rows or falsy result.

BigQueryCheckOperator(
    task_id="check_row_count",
    sql="SELECT COUNT(*) FROM `project.dataset.table` WHERE date = '{{ ds }}'",
    use_legacy_sql=False
)

BigQueryValueCheckOperator Operator

Use case: Validate query result matches expected value (within tolerance).

BigQueryValueCheckOperator(
    task_id="check_total",
    sql="SELECT SUM(amount) FROM `project.dataset.sales`",
    pass_value=1000000,
    tolerance=0.1,  # 10% tolerance
    use_legacy_sql=False
)

BigQueryCreateEmptyTableOperator Operator

Use case: Create a table with schema before loading data.

BigQueryCreateEmptyTableOperator(
    task_id="create_table",
    dataset_id="my_dataset",
    table_id="new_table",
    schema_fields=[
        {"name": "id", "type": "INTEGER", "mode": "REQUIRED"},
        {"name": "name", "type": "STRING", "mode": "NULLABLE"},
    ]
)

BigQueryDeleteTableOperator Operator

Use case: Drop temp/staging tables after pipeline completes.

BigQueryDeleteTableOperator(
    task_id="delete_staging",
    deletion_dataset_table="project.dataset.staging_table",
    ignore_if_missing=True
)

BigQueryCreateExternalTableOperator Operator

Use case: Create external table on top of GCS files (no data movement).

BigQueryCreateExternalTableOperator(
    task_id="create_external",
    table_resource={
        "tableReference": {"tableId": "ext_table", "datasetId": "raw"},
        "externalDataConfiguration": {
            "sourceUris": ["gs://bucket/data/*.parquet"],
            "sourceFormat": "PARQUET"
        }
    }
)

BigQueryTableExistenceSensor Sensor

Use case: Wait for an upstream table to be created before proceeding.

from airflow.providers.google.cloud.sensors.bigquery import BigQueryTableExistenceSensor

BigQueryTableExistenceSensor(
    task_id="wait_for_table",
    project_id="my-project",
    dataset_id="my_dataset",
    table_id="upstream_table",
    poke_interval=60,
    timeout=3600
)

GCS (Google Cloud Storage) Operators

GCSToBigQueryOperator Operator

Use case: Load files from GCS directly into BigQuery.

from airflow.providers.google.cloud.transfers.gcs_to_bigquery import GCSToBigQueryOperator

GCSToBigQueryOperator(
    task_id="load_to_bq",
    bucket="my-bucket",
    source_objects=["data/{{ ds }}/*.csv"],
    destination_project_dataset_table="project.dataset.table",
    source_format="CSV",
    skip_leading_rows=1,
    write_disposition="WRITE_TRUNCATE",
    autodetect=True
)

GCSToGCSOperator Operator

Use case: Move/copy files between buckets (e.g., raw → processed).

from airflow.providers.google.cloud.transfers.gcs_to_gcs import GCSToGCSOperator

GCSToGCSOperator(
    task_id="archive_files",
    source_bucket="raw-bucket",
    source_object="incoming/*.csv",
    destination_bucket="archive-bucket",
    destination_object="processed/{{ ds }}/",
    move_object=True  # deletes source after copy
)

GCSObjectExistenceSensor Sensor

Use case: Wait for a file to land in GCS before processing.

from airflow.providers.google.cloud.sensors.gcs import GCSObjectExistenceSensor

GCSObjectExistenceSensor(
    task_id="wait_for_file",
    bucket="landing-bucket",
    object="daily_feed/{{ ds }}/data.csv",
    poke_interval=300,
    timeout=7200,
    mode="reschedule"  # free up worker slot while waiting
)

GCSObjectsWithPrefixExistenceSensor Sensor

Use case: Wait for any files with a given prefix (e.g., _SUCCESS marker).

GCSObjectsWithPrefixExistenceSensor(
    task_id="wait_for_success",
    bucket="output-bucket",
    prefix="results/{{ ds }}/_SUCCESS",
    poke_interval=120
)

GCSListObjectsOperator Operator

Use case: List files in a bucket path, push to XCom for dynamic processing.

GCSListObjectsOperator(
    task_id="list_files",
    bucket="my-bucket",
    prefix="incoming/{{ ds }}/",
    delimiter=".csv"
)

GCSCreateBucketOperator Operator

Use case: Create a bucket as part of environment setup.

GCSCreateBucketOperator(
    task_id="create_bucket",
    bucket_name="my-new-bucket",
    storage_class="STANDARD",
    location="US"
)

GCSDeleteObjectsOperator Operator

Use case: Clean up temp files after processing.

GCSDeleteObjectsOperator(
    task_id="cleanup",
    bucket_name="temp-bucket",
    prefix="staging/{{ ds }}/"
)

GCSHook Hook

Use case: Programmatic GCS access inside PythonOperator.

from airflow.providers.google.cloud.hooks.gcs import GCSHook

def upload_result(**context):
    hook = GCSHook(gcp_conn_id="google_cloud_default")
    hook.upload(
        bucket_name="my-bucket",
        object_name="output.json",
        data='{"status":"done"}'
    )

PythonOperator(task_id="upload", python_callable=upload_result)

Dataproc Operators

DataprocCreateClusterOperator Operator

Use case: Spin up an ephemeral Spark cluster.

from airflow.providers.google.cloud.operators.dataproc import DataprocCreateClusterOperator

DataprocCreateClusterOperator(
    task_id="create_cluster",
    project_id="my-project",
    region="us-central1",
    cluster_name="etl-cluster-{{ ds_nodash }}",
    cluster_config={
        "master_config": {"num_instances": 1, "machine_type_uri": "n1-standard-4"},
        "worker_config": {"num_instances": 2, "machine_type_uri": "n1-standard-4"},
        "software_config": {"image_version": "2.1-debian11"}
    }
)

DataprocSubmitJobOperator Operator

Use case: Submit PySpark/Spark/Hive jobs to a Dataproc cluster.

from airflow.providers.google.cloud.operators.dataproc import DataprocSubmitJobOperator

DataprocSubmitJobOperator(
    task_id="run_spark_job",
    project_id="my-project",
    region="us-central1",
    job={
        "reference": {"project_id": "my-project"},
        "placement": {"cluster_name": "etl-cluster-{{ ds_nodash }}"},
        "pyspark_job": {
            "main_python_file_uri": "gs://scripts-bucket/etl/transform.py",
            "args": ["--date", "{{ ds }}"],
            "jar_file_uris": ["gs://jars-bucket/connector.jar"]
        }
    }
)

DataprocDeleteClusterOperator Operator

Use case: Tear down cluster after job completes (cost control).

from airflow.providers.google.cloud.operators.dataproc import DataprocDeleteClusterOperator

DataprocDeleteClusterOperator(
    task_id="delete_cluster",
    project_id="my-project",
    region="us-central1",
    cluster_name="etl-cluster-{{ ds_nodash }}",
    trigger_rule="all_done"  # delete even if job failed
)

DataprocCreateBatchOperator Operator

Use case: Run serverless Spark (no cluster management needed).

from airflow.providers.google.cloud.operators.dataproc import DataprocCreateBatchOperator

DataprocCreateBatchOperator(
    task_id="serverless_spark",
    project_id="my-project",
    region="us-central1",
    batch_id="batch-{{ ds_nodash }}",
    batch={
        "pyspark_batch": {
            "main_python_file_uri": "gs://bucket/job.py",
            "args": ["--date", "{{ ds }}"]
        },
        "runtime_config": {"version": "2.0"}
    }
)

Dataflow Operators

DataflowStartFlexTemplateOperator Operator

Use case: Run a Flex Template (custom container-based Dataflow job).

from airflow.providers.google.cloud.operators.dataflow import DataflowStartFlexTemplateOperator

DataflowStartFlexTemplateOperator(
    task_id="run_flex_template",
    project_id="my-project",
    location="us-central1",
    body={
        "launchParameter": {
            "containerSpecGcsPath": "gs://templates/my-template.json",
            "jobName": "etl-{{ ds_nodash }}",
            "parameters": {
                "input": "gs://input/{{ ds }}/",
                "output": "project:dataset.table"
            }
        }
    }
)

DataflowTemplatedJobStartOperator Operator

Use case: Run a classic Dataflow template (e.g., GCS-to-BQ).

from airflow.providers.google.cloud.operators.dataflow import DataflowTemplatedJobStartOperator

DataflowTemplatedJobStartOperator(
    task_id="run_template",
    template="gs://dataflow-templates/latest/GCS_Text_to_BigQuery",
    project_id="my-project",
    location="us-central1",
    parameters={
        "inputFilePattern": "gs://bucket/data/*.json",
        "outputTable": "project:dataset.table",
        "JSONPath": "gs://bucket/schema.json"
    }
)

PubSub Operators

PubSubPublishMessageOperator Operator

Use case: Publish messages to trigger downstream systems.

from airflow.providers.google.cloud.operators.pubsub import PubSubPublishMessageOperator

PubSubPublishMessageOperator(
    task_id="publish_event",
    project_id="my-project",
    topic="pipeline-events",
    messages=[{"data": b'{"status":"complete","date":"{{ ds }}"}'}]
)

PubSubPullSensor Sensor

Use case: Wait for a message before starting processing.

from airflow.providers.google.cloud.sensors.pubsub import PubSubPullSensor

PubSubPullSensor(
    task_id="wait_for_trigger",
    project_id="my-project",
    subscription="my-subscription",
    max_messages=1,
    poke_interval=60,
    mode="reschedule"
)

Core / Utility Operators

PythonOperator Operator

Use case: Run arbitrary Python logic (API calls, transformations, validations).

from airflow.operators.python import PythonOperator

PythonOperator(
    task_id="process_data",
    python_callable=my_function,
    op_kwargs={"date": "{{ ds }}"}
)

BranchPythonOperator Operator

Use case: Conditional branching based on runtime logic.

from airflow.operators.python import BranchPythonOperator

def choose_branch(**context):
    if context['params']['env'] == 'prod':
        return 'full_load_task'
    return 'incremental_task'

BranchPythonOperator(task_id="branch", python_callable=choose_branch)

ShortCircuitOperator Operator

Use case: Skip all downstream tasks if condition is False.

from airflow.operators.python import ShortCircuitOperator

ShortCircuitOperator(
    task_id="check_if_holiday",
    python_callable=lambda: not is_holiday(date.today())
)

TriggerDagRunOperator Operator

Use case: Trigger another DAG (cross-DAG orchestration).

from airflow.operators.trigger_dagrun import TriggerDagRunOperator

TriggerDagRunOperator(
    task_id="trigger_downstream",
    trigger_dag_id="downstream_dag",
    conf={"source_date": "{{ ds }}"},
    wait_for_completion=True,
    poke_interval=60
)

ExternalTaskSensor Sensor

Use case: Wait for a task in another DAG to complete.

from airflow.sensors.external_task import ExternalTaskSensor
from datetime import timedelta

ExternalTaskSensor(
    task_id="wait_for_upstream",
    external_dag_id="upstream_dag",
    external_task_id="final_task",
    execution_delta=timedelta(hours=1),
    mode="reschedule"
)

BashOperator Operator

Use case: Run shell commands, scripts, or CLI tools.

from airflow.operators.bash import BashOperator

BashOperator(
    task_id="run_script",
    bash_command="gsutil cp gs://bucket/file.csv /tmp/ && wc -l /tmp/file.csv"
)

EmptyOperator (formerly DummyOperator) Operator

Use case: DAG structure placeholder, join point after branching.

from airflow.operators.empty import EmptyOperator

EmptyOperator(task_id="start")
EmptyOperator(task_id="end", trigger_rule="none_failed_min_one_success")

Exceptions & State

AirflowFailException Exception

Use case: Immediately fail a task without retries.

from airflow.exceptions import AirflowFailException

def validate(**context):
    if row_count == 0:
        raise AirflowFailException("No data found - failing without retry")

AirflowSkipException Exception

Use case: Mark task as skipped programmatically.

from airflow.exceptions import AirflowSkipException

def conditional_run(**context):
    if not data_available():
        raise AirflowSkipException("No data today, skipping")

DagRunState Utility

Use case: Check/set DAG run state programmatically.

from airflow.utils.state import DagRunState
from airflow.models import DagRun

dag_runs = DagRun.find(dag_id="my_dag", state=DagRunState.FAILED)

Typical DAG Pattern (GCP Data Engineering)

start >> wait_for_file >> create_cluster >> submit_spark_job >> delete_cluster

delete_cluster >> load_to_bq >> check_data_quality >> publish_event >> end

Complete DAG Example

from airflow import DAG
from airflow.operators.empty import EmptyOperator
from airflow.providers.google.cloud.sensors.gcs import GCSObjectExistenceSensor
from airflow.providers.google.cloud.operators.dataproc import (
    DataprocCreateClusterOperator,
    DataprocSubmitJobOperator,
    DataprocDeleteClusterOperator,
)
from airflow.providers.google.cloud.transfers.gcs_to_bigquery import GCSToBigQueryOperator
from airflow.providers.google.cloud.operators.bigquery import BigQueryCheckOperator
from airflow.providers.google.cloud.operators.pubsub import PubSubPublishMessageOperator
from datetime import datetime

with DAG(
    dag_id="gcp_etl_pipeline",
    start_date=datetime(2024, 1, 1),
    schedule_interval="@daily",
    catchup=False,
    default_args={"retries": 2, "retry_delay": timedelta(minutes=5)}
) as dag:

    start = EmptyOperator(task_id="start")

    wait_for_file = GCSObjectExistenceSensor(
        task_id="wait_for_file",
        bucket="landing-bucket",
        object="feed/{{ ds }}/data.parquet",
        mode="reschedule"
    )

    create_cluster = DataprocCreateClusterOperator(
        task_id="create_cluster",
        project_id="my-project",
        region="us-central1",
        cluster_name="etl-{{ ds_nodash }}",
        cluster_config={...}
    )

    spark_job = DataprocSubmitJobOperator(
        task_id="spark_transform",
        project_id="my-project",
        region="us-central1",
        job={...}
    )

    delete_cluster = DataprocDeleteClusterOperator(
        task_id="delete_cluster",
        project_id="my-project",
        region="us-central1",
        cluster_name="etl-{{ ds_nodash }}",
        trigger_rule="all_done"
    )

    load_bq = GCSToBigQueryOperator(
        task_id="load_to_bq",
        bucket="processed-bucket",
        source_objects=["output/{{ ds }}/*.parquet"],
        destination_project_dataset_table="project.dataset.table",
        source_format="PARQUET",
        write_disposition="WRITE_TRUNCATE"
    )

    quality_check = BigQueryCheckOperator(
        task_id="quality_check",
        sql="SELECT COUNT(*) > 0 FROM `project.dataset.table` WHERE date='{{ ds }}'",
        use_legacy_sql=False
    )

    notify = PubSubPublishMessageOperator(
        task_id="notify",
        project_id="my-project",
        topic="pipeline-complete",
        messages=[{"data": b'{"dag":"gcp_etl","date":"{{ ds }}"}'}]
    )

    end = EmptyOperator(task_id="end")

    # DAG Dependencies
    start >> wait_for_file >> create_cluster >> spark_job >> delete_cluster
    delete_cluster >> load_bq >> quality_check >> notify >> end
